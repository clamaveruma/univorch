% ===================== Terms =====================
% --- Descriptor tree ---
\nomenclature[T-descriptor]{\textbf{Descriptor}}{The orchestrator object
  representing a VM (its definition plus its state). Analogous to a file
  descriptor in an operating system.}
\nomenclature[T-folder]{\textbf{Folder}}{Grouping node in the tree,
  container of descriptors and other folders. Supports explicit import of
  inheritable resources.}
\nomenclature[T-matpath]{\textbf{Materialized path}}{Unique identifier of
  a folder or descriptor; the full path from the root (e.g.\
  \texttt{/lab/networks/student01}).}

% --- Inheritable resources ---
\nomenclature[T-template]{\textbf{Template}}{Resource declarable with
  \texttt{define templates:}; offers reusable default values that
  descriptors reference with \texttt{use template:}.}
\nomenclature[T-hypervisor]{\textbf{Hypervisor (resource)}}{Resource
  declarable with \texttt{define hypervisors:}; names a specific
  hypervisor instance (with its type and connection data) that
  descriptors and templates reference with \texttt{use hypervisor:}.}
\nomenclature[T-import]{\textbf{Import}}{Mechanism by which a folder
  states which resources from its parent become visible here; without
  import, the resource is not accessible from this branch.}
\nomenclature[T-closure]{\textbf{Lexical closure}}{Rule by which internal
  references inside a template are resolved from where the template is
  defined, not from where it is used.}

% --- State ---
\nomenclature[T-state]{\textbf{State}}{Descriptor state axis from the
  orchestrator's point of view: \texttt{provisioned}, \texttt{deployed}
  (with or without \texttt{drifted}), \texttt{broken},
  \texttt{unreachable}.}
\nomenclature[T-runtime]{\textbf{Runtime}}{VM state axis from the
  hypervisor's point of view: \texttt{running}, \texttt{stopped},
  \texttt{paused}, \texttt{unknown}.}

% --- Operations ---
\nomenclature[T-job]{\textbf{Job}}{Persistent record of an operation,
  with its unique identifier, type, target, status and result. Provides
  audit and prepares for future asynchronous execution.}
\nomenclature[T-command]{\textbf{Command}}{Design pattern applied to
  operations: each action encapsulates \texttt{validate()} and
  \texttt{execute()}, enabling dry-run (\texttt{plan}), deferred
  execution and retry.}
\nomenclature[T-connector]{\textbf{Connector}}{Adapter between the
  orchestrator and a concrete hypervisor. Implements the
  \texttt{HypervisorConnector} ABC contract.}

% --- User actions ---
\nomenclature[T-load]{\textbf{load}}{Action that loads a YAML definition
  document into a destination folder of the tree.}
\nomenclature[T-deploy]{\textbf{deploy / undeploy}}{Create or destroy the
  VM on the hypervisor from a descriptor (transition between
  \texttt{provisioned} and \texttt{deployed}).}
\nomenclature[T-start]{\textbf{start / stop}}{Power on or off the VM that
  is already deployed (changes the runtime axis, not the state axis).}
